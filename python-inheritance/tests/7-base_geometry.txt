bg = BaseGeometry()

# Valid integers
bg.integer_validator("width", 5)
bg.integer_validator("height", 12)

# Invalid types
try:
    bg.integer_validator("name", "John")
except Exception as e:
    print(e)  # TypeError: name must be an integer

try:
    bg.integer_validator("age", [3])
except Exception as e:
    print(e)  # TypeError: age must be an integer

try:
    bg.integer_validator("distance", None)
except Exception as e:
    print(e)  # TypeError: distance must be an integer

# Invalid integer values
try:
    bg.integer_validator("age", 0)
except Exception as e:
    print(e)  # ValueError: age must be greater than 0

try:
    bg.integer_validator("distance", -4)
except Exception as e:
    print(e)  # ValueError: distance must be greater than 0
